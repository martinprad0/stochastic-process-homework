\[ f(0) = [0,1,2,3] \to [1,0,2,3] = f(6),  \]
\[ f(0) = [0,1,2,3] \to [1,2,0,3] = f(8),  \]
\[ f(0) = [0,1,2,3] \to [1,2,3,0] = f(9).  \]

\[ \Pi_{0,6} = \Pi_{0,8} = \Pi_{0,9} = 1/3, \]
